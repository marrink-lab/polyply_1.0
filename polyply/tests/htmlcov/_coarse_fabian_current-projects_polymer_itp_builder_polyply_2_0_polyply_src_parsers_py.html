34 def treat_link_multiple(self): 

35 """ 

36 Iterates over a :class:`vermouth.force_field.ForceField` and 

37 adds version tags for all interactions within a 

38 :class:`vermouth.molecule.Link` that are applied to the same atoms. 

39 """ 

40 for link in self.force_field.links: 

41 for key in link.interactions: 

42 terms = link.interactions[key] 

43 count_terms = Counter(tuple(term.atoms) for term in terms) 

44 for term in terms: 

45 tag = count_terms[tuple(term.atoms)] 

46 if tag >= 1: 

47 term.meta.update({"version":tag}) 

48 count_terms[tuple(term.atoms)] = tag -1

50 def _treat_link_atoms(self, block, link, inter_type): 

51 

52 # we need to convert the atom index to an atom-name
